function double(var x) {
	var y=13;
	if(*x>*y) {
		return 2*(*x);
	}
	else {
		return *y;
	}
}
function main() {
	var x=1;
	while(double(*x) == 2*(*x)) {
		*x=*x+1;
	}
	return *x;
}
