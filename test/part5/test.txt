function main() {
    var x = read();
    x=x+1;
    var y=3;
    y=y+5;
    var sum1=0;
    for(i=1:10) {
        sum1=sum1+i;
    }
    var sum2=0;
    while(x<y) {
        sum2=sum2+x;
        if(y%x==0) {
            x=x+1;
        } else {
            y=y-1;
        }
    }
    print(sum1+sum2);
    return sum1+sum2;
}
