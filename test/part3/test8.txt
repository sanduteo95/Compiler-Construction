/* Testing:
	- tuples
	- returning a tuple from a function
	- operations on tuples
	- tuple as a function argument
*/
function create(var x, var y, var z) {
	var tuple = (x, y, z);
	return *tuple;
}
function compare(var p, var q) {
	return *p < *q;
}
function main() {
	var p = create(1, 2, "string1");
	var q = create(2, 3, "string2");
	return compare(*p, *q);
}
